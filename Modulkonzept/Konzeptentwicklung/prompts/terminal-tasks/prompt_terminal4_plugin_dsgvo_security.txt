     plugin: PluginRegistryRow,

     payload: unknown,
   ): Promise<PluginExecutionResult> {

     const body = JSON.stringify({ event: hookEvent, data: payload, timestamp: new Date().toISOString() });
     const sig = hmacSha256(plugin.webhook_secret, body); // nutze crypto.createHmac

     try {
       const res = await fetch(plugin.webhook_url, {
         method: 'POST',
         headers: {
           'Content-Type': 'application/json',
           'X-ProzessPilot-Signature': sig,
           'X-ProzessPilot-Event': hookEvent,
         },
         body,
         signal: AbortSignal.timeout(10_000), // 10s Timeout
       });
       const responseBody = await res.text().catch(() => '');

ERSTELLE FOLGENDE DATEIEN (keine neuen npm packages nötig, reine Shell/Python-Skripte):

1. /infra/backup/backup-postgres.sh
   ```bash
   #!/bin/bash
   # ProzessPilot PostgreSQL Backup
   # Cron: 0 2 * * * /infra/backup/backup-postgres.sh >> /var/log/pp-backup.log 2>&1

   set -euo pipefail
   DATE=$(date +%Y%m%d_%H%M%S)
   BACKUP_DIR="${BACKUP_DIR:-/backups/postgres}"
   DB_URL="${DATABASE_URL:-postgresql://localhost:5432/prozesspilot}"
   RETENTION_DAYS="${RETENTION_DAYS:-90}"

   mkdir -p "$BACKUP_DIR"
   FILENAME="prozesspilot_${DATE}.sql.gz"

   echo "[$(date -Is)] Starting backup: $FILENAME"
   pg_dump "$DB_URL" | gzip > "$BACKUP_DIR/$FILENAME"
   echo "[$(date -Is)] Backup complete: $(du -h "$BACKUP_DIR/$FILENAME" | cut -f1)"

   # Alte Backups löschen (älter als RETENTION_DAYS)
   find "$BACKUP_DIR" -name "*.sql.gz" -mtime +$RETENTION_DAYS -delete
   echo "[$(date -Is)] Cleanup done. Files remaining: $(ls "$BACKUP_DIR" | wc -l)"

   # Optional: S3-Upload (wenn AWS_BACKUP_BUCKET gesetzt)
   if [ -n "${AWS_BACKUP_BUCKET:-}" ]; then
     aws s3 cp "$BACKUP_DIR/$FILENAME" "s3://$AWS_BACKUP_BUCKET/postgres/$FILENAME"
     echo "[$(date -Is)] Uploaded to S3: s3://$AWS_BACKUP_BUCKET/postgres/$FILENAME"
   fi
   ```

2. /infra/backup/backup-s3.sh
   ```bash
   #!/bin/bash
   # ProzessPilot S3/MinIO Archiv-Backup (Sync zu Backup-Bucket)
   # Cron: 0 3 * * * /infra/backup/backup-s3.sh >> /var/log/pp-backup.log 2>&1

   set -euo pipefail
   SOURCE_BUCKET="${PP_S3_BUCKET:-prozesspilot-receipts}"
   BACKUP_BUCKET="${AWS_BACKUP_BUCKET:-prozesspilot-backup}"
   S3_ENDPOINT="${PP_S3_ENDPOINT:-}"  # leer = AWS, sonst MinIO

   ENDPOINT_FLAG=""
   if [ -n "$S3_ENDPOINT" ]; then
     ENDPOINT_FLAG="--endpoint-url $S3_ENDPOINT"
   fi

   echo "[$(date -Is)] Starting S3 sync: $SOURCE_BUCKET → $BACKUP_BUCKET/s3-archive/"
   aws s3 sync $ENDPOINT_FLAG "s3://$SOURCE_BUCKET" "s3://$BACKUP_BUCKET/s3-archive/" --storage-class STANDARD_IA
   echo "[$(date -Is)] S3 sync complete"
   ```

3. /infra/backup/restore-test.sh
   ```bash
   #!/bin/bash
   # Monatlicher Restore-Test (letztes Backup in Test-DB laden)
   # Cron: 0 4 1 * * /infra/backup/restore-test.sh >> /var/log/pp-restore-test.log 2>&1

   set -euo pipefail
   BACKUP_DIR="${BACKUP_DIR:-/backups/postgres}"
   TEST_DB_URL="${TEST_DB_URL:-postgresql://localhost:5432/prozesspilot_restore_test}"

   LATEST=$(ls -t "$BACKUP_DIR"/*.sql.gz 2>/dev/null | head -1)
   if [ -z "$LATEST" ]; then
     echo "[$(date -Is)] ERROR: Kein Backup gefunden in $BACKUP_DIR"
     exit 1
   fi

   echo "[$(date -Is)] Restore-Test mit: $LATEST"
   # Test-DB neu erstellen
   psql "${TEST_DB_URL%/*}" -c "DROP DATABASE IF EXISTS prozesspilot_restore_test;" 2>/dev/null || true
   psql "${TEST_DB_URL%/*}" -c "CREATE DATABASE prozesspilot_restore_test;"
   gunzip -c "$LATEST" | psql "$TEST_DB_URL"

   # Einfacher Sanity-Check
   COUNT=$(psql "$TEST_DB_URL" -tAc "SELECT COUNT(*) FROM receipts;" 2>/dev/null || echo "ERROR")
   echo "[$(date -Is)] Restore-Test OK. Receipts in Backup: $COUNT"
   ```

════════════════════════════════════════════════════════════════

Ziel: OWASP Top 10 durchgehen und Schwachstellen beheben.

1. Lies die Backend-Codebase: /backend/src/
   Prüfe konkret diese Punkte und BEHEBE gefundene Probleme direkt im Code:

   A) SQL Injection:
   - Stelle sicher, dass ALLE DB-Queries parametrisierte Queries nutzen (kein String-Concat mit User-Input)
   - Suche nach Pattern wie: `query("... WHERE id = '" + variable`
   - Falls gefunden: ersetze mit $1, $2 Parameter-Syntax

   B) Rate Limiting:
   - Prüfe ob fastify-rate-limit bereits installiert ist
   - Falls ja: stelle sicher dass alle öffentlichen Endpunkte (besonders /api/v1/receipts POST) Rate-Limits haben
   - Falls nein: installiere (`npm install @fastify/rate-limit`) und konfiguriere in server.ts oder app.ts:
     ```ts
     await app.register(import('@fastify/rate-limit'), {
       global: true,
       max: 100,
       timeWindow: '1 minute',
       keyGenerator: (req) => req.headers['x-tenant-id'] as string || req.ip,
     });
     ```

   C) Secrets Handling:
   - Stelle sicher dass keine Secrets hardcodiert sind (grep nach 'password', 'secret', 'apikey' in src/)
   - Alle ENV-Variablen sollten in /backend/.env.example dokumentiert sein
   - Erstelle /backend/.env.example falls nicht vorhanden mit allen benötigten Variablen (keine echten Werte!)
   - Stelle sicher dass .env in .gitignore ist

   D) Input Validation:
   - Prüfe ob alle Route-Handler Zod-Schemas für Body, Params, Query nutzen
   - Fehlende Validierungen ergänzen (besonders bei PATCH/PUT Routen)

   E) HMAC-Stärke:
   - Prüfe HMAC-Implementation: muss SHA256 oder stärker sein
   - Timing-sichere Vergleiche: crypto.timingSafeEqual() statt ===

   F) Error Responses:
   - Stelle sicher dass Error-Responses keine Stack-Traces oder DB-Fehlermeldungen nach außen leaken
   - In Fastify: setErrorHandler, der in Produktion nur { error: 'Internal Server Error' } zurückgibt

     const hostname = new URL(url).hostname;
     // Einfache Prüfung: 127.x, 10.x, 192.168.x, 172.16-31.x
     return /^(127\.|10\.|192\.168\.|172\.(1[6-9]|2\d|3[01])\.)/.test(hostname) ||
