4. Datei: /infra/load-tests/scenarios/upload-receipts.js
   ```js
   import http from 'k6/http';
   import { check, sleep } from 'k6';
   import { Counter, Trend } from 'k6/metrics';
   import { BASE_URL, TENANT_ID, DEFAULT_OPTIONS } from '../config.js';

   export const options = {
     ...DEFAULT_OPTIONS,
     scenarios: {
       concurrent_uploads: {
         executor: 'ramping-vus',
         startVUs: 0,
         stages: [
           { duration: '30s', target: 100 },   // Hochfahren auf 100 VUs
           { duration: '60s', target: 100 },   // 100 VUs halten
           { duration: '20s', target: 0 },     // Runterfahren
         ],
       },
     },
   };

   const uploadErrors   = new Counter('upload_errors');
   const processingTime = new Trend('processing_time_ms');

   // Minimales Test-PDF als Base64 (1-KB Dummy)
   const DUMMY_PDF_B64 = 'JVBERi0xLjMKJcfs...'; // gültiges Minimal-PDF

   export default function () {
     const payload = {
       file:        http.file(open('../fixtures/test-receipt.pdf'), 'test-receipt.pdf', 'application/pdf'),
       customer_id: 'load-test-customer',
     };

     const headers = { 'X-Tenant-ID': TENANT_ID };
     const start   = Date.now();
     const res     = http.post(`${BASE_URL}/api/v1/receipts`, payload, { headers });

     processingTime.add(Date.now() - start);

     const ok = check(res, {
       'status 201': (r) => r.status === 201,
       'hat receipt_id': (r) => JSON.parse(r.body)?.receipt_id !== undefined,
     });

     if (!ok) uploadErrors.add(1);

     sleep(Math.random() * 2); // 0–2s Pause zwischen Requests
   }
   ```

2. /infra/runbook/01_deployment.md — Deployment-Anleitung
   Inhalte:
   - Voraussetzungen (Node 20+, PostgreSQL 15+, n8n, MinIO/S3)
   - Umgebungsvariablen Checkliste (alle .env Werte mit Beschreibung)
   - Schritt-für-Schritt Erstinstallation:
     1. Repository klonen
     2. pnpm install
     3. .env konfigurieren
     4. PostgreSQL Datenbank anlegen
     5. Migrationen ausführen: npm run migrate
     6. Backend starten: npm run start
     7. n8n starten + Workflows importieren
     8. Webapp bauen + deployen: npm run build

   - Nginx-Config Beispiel (Reverse Proxy für Backend + Webapp)

3. /infra/runbook/02_rollback.md — Rollback-Anleitung
   Inhalte:
   - Wann rollback nötig ist
   - DB-Rollback: letztes pg_dump backup einspielen (Befehl)
   - Code-Rollback: git revert / git checkout previous tag
   - n8n Workflow-Rollback: JSON-Export vor Deployment machen
   - Smoke-Test nach Rollback: welche Endpunkte prüfen?

4. /infra/runbook/03_oncall_playbook.md — On-Call Playbook
   Format pro Szenario: Symptom → Diagnose-Befehle → Lösung

   Szenarien:
   - Backend antwortet nicht (502/504)
     Diagnose: pm2 status, docker ps, netstat -tlnp :3000
     Lösung: pm2 restart pp-backend / docker-compose restart backend
   - Datenbank-Verbindung verloren
     Diagnose: pg_isready, tail /var/log/postgresql/...
     Lösung: service postgresql restart, Connection-Pool prüfen
   - n8n Workflow läuft nicht durch
     Diagnose: n8n UI → Executions → letzten Fehler anschauen
     Lösung: Workflow manuell triggern, Webhook-URL prüfen
   - Beleg steckt in Status "extracting" fest
     Diagnose: SELECT * FROM receipts WHERE status='extracting' AND updated_at < now()-interval '10min'
     Lösung: UPDATE receipts SET status='received' WHERE ... (re-trigger)
   - Hohe Fehlerrate bei Kategorisierung
     Diagnose: SELECT error_type, COUNT(*) FROM receipt_errors GROUP BY 1
     Lösung: Claude API-Key prüfen, Rate-Limit kontrollieren
   - S3/MinIO nicht erreichbar
     Diagnose: aws s3 ls s3://bucket-name --endpoint-url ...
     Lösung: MinIO-Service neu starten, S3-Credentials prüfen

5. /infra/runbook/04_tenant_onboarding.md — Tenant-Onboarding-Checkliste
   Checkliste für jeden neuen Kunden:
   - [ ] Tenant in DB anlegen: INSERT INTO tenants ...
   - [ ] Customer Profile anlegen mit korrekten enabled_modules
   - [ ] WhatsApp-Nummer in customer_profiles eintragen
   - [ ] Lexoffice/sevDesk API-Key in customer_credentials speichern
   - [ ] Test-Beleg über WhatsApp senden und Pipeline durchlaufen lassen

   - [ ] Monatsreport-Cron für den Kunden aktivieren
   - [ ] DSGVO-Einwilligung + AV-Vertrag unterschrieben?
   - [ ] Backup überprüfen (customer_id taucht in neuem pg_dump auf?)
   - [ ] Onboarding-Mail mit Zugangsdaten versenden

6. /infra/runbook/05_monitoring_checks.md — Tägliche/Wöchentliche Checks
   Tägliche Checks (5 Min):
   - Backup-Log: tail /var/log/pp-backup.log | grep ERROR

   Wöchentliche Checks (15 Min):
   - npm audit im Backend: keine critical vulnerabilities
   - DB-Größe: SELECT pg_size_pretty(pg_database_size('prozesspilot'))
   - Restore-Test Logs prüfen
   - Load-Test einmal kurz ausführen (10 VUs, 30s)
